function fearNotLetter(str) {
  let firstValue = str.charCodeAt(0);
  let secondValue = 0;
  let result = undefined;
  for (let i = 1; i < str.length; i++) {
    secondValue = str.charCodeAt(i)
    if ((secondValue - firstValue) > 1) {
      secondValue -= 1
      result = String.fromCharCode(secondValue)
      break
    } else {
      firstValue++
    }
  }
  return result;
}

fearNotLetter("abce");
