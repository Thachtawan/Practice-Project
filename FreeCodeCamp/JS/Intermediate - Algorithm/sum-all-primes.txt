function checkPrime(number) {
  if (number === 0 || number === 1) {
    return false;
  } else {
    for (let i = 2; i <= number; i++) {
      if (number % i === 0 && number != i) {
        return false;
      };
    } return true;
  }
};

function sumPrimes(num) {
  let result = 0;
  for (let number = 0; number <= num; number++) {
    if(checkPrime(number)) {
      result += number;
    }
  }
  return result;
}

sumPrimes(10);
