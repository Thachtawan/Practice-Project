function diffArray(arr1, arr2) {
  const newArr = [];
  arr1.forEach(value => {
      if (!arr2.includes(value)) {
          newArr.push(value)
      }
  })
  arr2.forEach(value => {
      if (!arr1.includes(value)) {
          newArr.push(value)
      }
  })
  return newArr;
}

diffArray([1, 2, 3, 5], [1, 2, 3, 4, 5]);
