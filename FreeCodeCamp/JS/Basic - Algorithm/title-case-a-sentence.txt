function titleCase(str) {
  let arr = str.split(" ")
  let newStr = "";
  for (let i = 0; i < arr.length; i++) {
    if (i == arr.length - 1) {
      newStr += arr[i][0].toUpperCase() + arr[i].slice(1).toLowerCase()
    } else {
      newStr += arr[i][0].toUpperCase() + arr[i].slice(1).toLowerCase() + " "
    }
  }
  return newStr;
}

titleCase("I'm a little tea pot");
